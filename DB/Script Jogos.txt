
CREATE TABLE Fabricantes (
id INT PRIMARY KEY IDENTITY(1,1),
nome NVARCHAR(100) NOT NULL
);

CREATE TABLE Publicadora (
ID INT PRIMARY KEY IDENTITY(1,1),
nome NVARCHAR(100) NOT NULL
)

CREATE TABLE Plataformas (
id INT PRIMARY KEY IDENTITY(1,1),
nome NVARCHAR(100) NOT NULL,
fabricante BIT NOT NULL,
fabricanteID INT,
CONSTRAINT FK_Plataformas_Fabricantes FOREIGN KEY (FabricanteID) REFERENCES Fabricantes(ID)
);


CREATE TABLE Generos (
id INT PRIMARY KEY IDENTITY(1,1),
nome NVARCHAR(100) NOT NULL,
descricao NVARCHAR(255)
);


CREATE TABLE Games (
id INT PRIMARY KEY IDENTITY(1,1),
titulo NVARCHAR(200) NOT NULL,
id_genero INT,
id_plataforma INT,
id_publicadora INT,
dataLancamento DATE,
Description NVARCHAR(1000),
CONSTRAINT FK_Games_Generos FOREIGN KEY (id_genero) REFERENCES Generos(id),
CONSTRAINT FK_Games_Plataformas FOREIGN KEY (id_plataforma) REFERENCES Plataformas(id),
CONSTRAINT FK_Games_Publicadora FOREIGN KEY (id_publicadora) REFERENCES Publicadora(id)
);


CREATE TABLE Locacao (
id INT PRIMARY KEY IDENTITY(1,1),
id_usuario INT,
id_game INT,
data_inicio DATE,
data_entrega DATE,
entregue BIT,
CONSTRAINT FK_Locacao_Usuario FOREIGN KEY (id_usuario) REFERENCES Usuario(id),
CONSTRAINT FK_Locacao_Games FOREIGN KEY (id_game) REFERENCES Games(id)
);
